#!/bin/bash

set -e

echo "=============================="
echo " ARCH LINUX SAFE INSTALLER"
echo " (DUAL BOOT SAFE MODE)"
echo "=============================="

echo ""
echo "Step 1: Disk overview (DO NOT SKIP)"
lsblk
echo ""
echo "⚠️ Identify carefully:"
echo "- EFI partition (FAT32, ~100–500MB)"
echo "- Linux ROOT partition (your empty D: space)"
echo "- NEVER select Windows NTFS partition"
echo ""

read -p "Enter EFI partition (e.g. /dev/nvme0n1p1): " EFI_PART
read -p "Enter ROOT partition (Linux only!): " ROOT_PART

echo ""
echo "You selected:"
echo "EFI  = $EFI_PART"
echo "ROOT = $ROOT_PART"
echo ""

read -p "Type YES to confirm (this will ERASE ROOT ONLY): " CONFIRM
if [ "$CONFIRM" != "YES" ]; then
    echo "Aborted."
    exit 1
fi

echo ""
echo "Formatting ROOT partition ONLY..."
mkfs.ext4 -F "$ROOT_PART"

echo "Mounting system..."
mount "$ROOT_PART" /mnt

mkdir -p /mnt/boot/efi
mount "$EFI_PART" /mnt/boot/efi

echo "Installing base system..."
pacstrap /mnt base linux linux-firmware linux-headers \
  networkmanager sudo nano git grub efibootmgr os-prober

genfstab -U /mnt >> /mnt/etc/fstab

arch-chroot /mnt /bin/bash <<EOF

echo "Setting up system..."

ln -sf /usr/share/zoneinfo/Asia/Manila /etc/localtime
hwclock --systohc

sed -i 's/#en_US.UTF-8/en_US.UTF-8/' /etc/locale.gen
locale-gen

echo "LANG=en_US.UTF-8" > /etc/locale.conf

echo "arch-linux" > /etc/hostname

systemctl enable NetworkManager

echo "Creating user..."
useradd -m -G wheel user
echo "user ALL=(ALL) ALL" >> /etc/sudoers

echo "Installing NVIDIA drivers..."
pacman -S --noconfirm nvidia nvidia-utils

echo "Installing Hyprland..."
pacman -S --noconfirm hyprland waybar kitty \
  pipewire pipewire-pulse wireplumber \
  xdg-desktop-portal-hyprland

echo "Gaming stack..."
pacman -S --noconfirm steam lutris wine winetricks mangohud gamemode

echo "Installing GRUB (dual boot safe)..."
pacman -S --noconfirm grub efibootmgr os-prober

sed -i 's/#GRUB_DISABLE_OS_PROBER=false/GRUB_DISABLE_OS_PROBER=false/' /etc/default/grub || true

grub-install --target=x86_64-efi --efi-directory=/boot/efi --bootloader-id=GRUB
grub-mkconfig -o /boot/grub/grub.cfg

EOF

echo ""
echo "INSTALL COMPLETE."
echo "Reboot and remove USB."
